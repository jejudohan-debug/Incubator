template <typename T>
void InputControl::updateConfigValue(T& configMember,
 void (DisplayState::*setter)(T), T newValue, UpdateFlag flag)
{
    configMember = newValue;         // 1. 임시 설정값 저장
    (_view.*setter)(newValue);       // 2. 뷰 데이터 업데이트
    _view.updateFlags |= flag;       // 3. 업데이트 플래그 추가
}

// 적용
void InputControl::increaseTemperature()
{
    float next = _view.getTargetTemp() + 0.5f;
    if (next <= 45.0f) {
        updateConfigValue(_tempConfig.targetTemp,
         &DisplayState::setTargetTemp, next, UpdateFlag::T_TEMP);
    }
}

void InputControl::increaseHumidity()
{
    float next = _view.getTargetHumi() + 1.0f;
    if (next <= 99.0f) {
        updateConfigValue(_tempConfig.targetHumi, &DisplayState::setTargetHumi, next, UpdateFlag::T_HUMI);
    }
}

void InputControl::increaseTurnInterval()
{
    uint16_t next = _view.getTurnInterval() + 1;
    if (next <= 720) {
        updateConfigValue(_tempConfig.turnInterval, &DisplayState::setTurnInterval, next, UpdateFlag::R_TURN);
    }
}
